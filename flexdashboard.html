NYC Restaurant Inspection Results

data("rest_inspec")

rest_inspec =
  rest_inspec |> 
  select(-action, -camis, -phone, -inspection_type) |> 
  filter(
    !is.na(grade),
    !is.na(violation_code),
    !is.na(boro),
    cuisine_description == "American")

Number of American Cuisine restaurants by borough.

rest_inspec |> 
  count(boro) |> 
  plot_ly(y = ~n, color = ~boro, type = "bar", colors = "viridis")

Score of restaurants serving American cuisine by borough.

rest_inspec |> 
  mutate(boro = fct_reorder(boro, score)) |> 
  plot_ly(y = ~score, color = ~boro, type = "box", colors = "viridis")
rest_inspec |> 
  group_by(boro, grade) |> 
  count(grade) |> 
  plot_ly(y = ~n, color = ~grade, type = "bar", colors = "viridis") |>
  layout(xaxis = list(type = 'category'))
